                input.value = '';

        // Modus wechseln

        window.toggleMode = function() {

            isEditMode = !isEditMode;

            const modeSwitch = document.getElementById('modeSwitch');

            const editMode = document.getElementById('editMode');

            const sendMode = document.getElementById('sendMode');

            const editLabel = document.getElementById('editLabel');

            const sendLabel = document.getElementById('sendLabel');

            

            if (navigator.vibrate) {

                navigator.vibrate(75);

            }

            

            if (isEditMode) {

                modeSwitch.classList.remove('active');

                editMode.classList.remove('hidden');

                sendMode.classList.remove('active');

                editLabel.classList.remove('inactive');

                sendLabel.classList.add('inactive');

            } else {

                modeSwitch.classList.add('active');

                editMode.classList.add('hidden');

                sendMode.classList.add('active');

                editLabel.classList.add('inactive');

                sendLabel.classList.remove('inactive');

            }

        };


        // Link-Funktionen

        window.copyLink = async function() {

            const link = document.getElementById('musicianLink').textContent;

            

            try {

                if (navigator.clipboard) {

                    await navigator.clipboard.writeText(link);

                    showCopyConfirmation();

                }

                

                if (navigator.vibrate) {

                    navigator.vibrate(100);

                }

            } catch (err) {

                console.log('Kopieren fehlgeschlagen');

            }

        };


        window.shareLink = async function() {

            const link = document.getElementById('musicianLink').textContent;

            

            try {

                if (navigator.share) {

                    await navigator.share({

                        title: 'Musiker-App',

                        text: 'Dirigent App für Musiker',

                        url: link

                    });

                } else {

                    copyLink();

                }

            } catch (err) {

                console.log('Teilen abgebrochen');

            }

        };


        function showCopyConfirmation() {

            const btn = document.querySelector('.copy-btn');

                btn.textContent = originalText;

                btn.style.background = '#007bff';

        }
